Temperature Conversion

This ruby program will prompt the user for a temperature in degrees Celsius and let the user know what the corresponding temperature is in Fahrenheit.

puts "Enter temperature in Celsius:"
degCels = gets.to_f

def getFahrenheit(degCels)
  sum = (degCels * 1.8) + 32
  return sum
end
valsum = getFahrenheit(degCels)
puts "#{degCels} degrees celsuis or #{valsum}F."

def ordinal(n)
  ending = case n % 100
           when 11, 12, 13 then 'th'
           else
             case n % 10
             when 1 then 'st'
             when 2 then 'nd'
             when 3 then 'rd'
             else 'th'
             end
           end

  puts "This is #{n}#{ending} item"
end
